# 오라클 컬럼 타입 조회
SELECT COLUMN_NAME,
       DATA_TYPE,
       DATA_LENGTH,
       DATA_PRECISION,
       DATA_SCALE
FROM ALL_TAB_COLUMNS
WHERE OWNER = 'VARDNA'
  AND TABLE_NAME = 'T_CUE'
  AND COLUMN_NAME = 'F_DATE';
  
# 내 스키마 테이블 목록 조회
SELECT
    table_name,
    column_name,
    data_type,
    data_length
FROM user_tab_columns
ORDER BY table_name, column_id;
